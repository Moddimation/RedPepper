bool isStep( IUseNerve* p, int step )
{
        return p->getNerveKeeper()->getStep() == step;
}

{
        return p->getNerveKeeper()->getCurrentNerve() == nerve;
}

{
        return p->getNerveKeeper()->getStep() == 0;
}

{
        return p->getNerveKeeper()->getStep() > t;
}

        return p->getNerveKeeper()->getStep() >= t;
}
