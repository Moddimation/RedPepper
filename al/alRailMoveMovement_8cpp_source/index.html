        tryGetArg( (int*)&mMoveType, getPlacementInfo( info ), moveTypeParamName );
        if ( mMoveType >= 2 )
                mMoveType = 0;

void RailMoveMovement::exeMove()
{
        if ( isExistRail( mHost ) )
        {
                if ( mMoveType == 0 )
                        moveSyncRail( mHost, mSpeed );
                else if ( mMoveType == 1 )
                        moveSyncRailTurn( mHost, mSpeed );
                else if ( mMoveType == 2 )
                        moveSyncRailLoop( mHost, mSpeed );
        }
}
